#Castilian Civil War
#A mix of the civil wars with outside intervention and noble families siding against the crown between 1419 and 1479.

LE_centralization_war = {
	potential = {
		check_variable {
			which = Noble_Militancy
			value = 50
		}
	}


	can_start = {
		check_variable {
			which = Noble_Militancy
			value = 75
		}
	}
	
	
	can_stop = {
		NOT = {
			check_variable {
				which = Noble_Militancy
				value = 50
			}
		}
	}
	
	progress = {
		modifier = {
			factor = 1
			NOT = { legitimacy = 90 }
		}
		modifier = {
			factor = 2
			check_variable {
				which = Noble_Militancy
				value = 75
			}
		}
		modifier = {
			factor = 3
			check_variable {
				which = Noble_Militancy
				value = 85
			}
		}
		modifier = {
			factor = 5
			check_variable {
				which = Noble_Militancy
				value = 100
			}
		}
		modifier = {
			factor = -1
			stability = 3
		}
	}




	
	can_end = {
		stability = 1
		NOT = { num_of_rebel_controlled_provinces = 1 }
		NOT = {
			has_spawned_rebels = noble_rebels
		}
	}
	



	modifier = {
		global_unrest = 5
	}	
	
	
	on_start = laws_events.5
	on_end = laws_events.6
	
	on_monthly = {
		events = {
		}
		random_events = { 
			11 = 0
			1 = laws_events.4
		}
	}
}
